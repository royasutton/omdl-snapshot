61 //! Return given value, if defined, or a secondary value, if primary is not defined.
62 /***************************************************************************//**
63  \param v <value> A primary value.
64  \param d <value> A secondary value.
65 
66  \returns <value> \p v when it is defined and \p d otherwise.
67 
68  The value \p d is returned only when \p v is equal to \b undef.
69 *******************************************************************************/
70 function defined_or
71 (
72  v,
73  d
74 ) = is_undef(v) ? d : v;

76 //! Return a circular index position.
77 /***************************************************************************//**
78  \param i <integer> A integer position.
79  \param l <integer> The range length.
80  \param f <integer> The starting index position.
81 
82  \returns <integer> The index position mapped into a circular list
83  within the range <tt>[f : l+f-1]</tt>.
84 *******************************************************************************/
85 function index_c
86 (
87  i,
88  l,
89  f = 0
90 ) = f + (((i % l) + l) % l);
